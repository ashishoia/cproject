#include <stdio.h>

struct Timestamp {
    int hours;
    int minutes;
    int seconds;
};

int main() {
    struct Timestamp timestamps[] = {
        {10, 30, 45},
        {12, 15, 20},
        {16, 45, 50}
    };
    int size = sizeof(timestamps) / sizeof(timestamps[0]);

    for (int i = 0; i < size; i++) {
        printf("Timestamp %d: %02d:%02d:%02d\n", i+1, timestamps[i].hours, timestamps[i].minutes, timestamps[i].seconds);
    }

    return 0;
}
